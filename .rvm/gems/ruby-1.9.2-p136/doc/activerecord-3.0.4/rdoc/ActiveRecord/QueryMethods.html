234:       association_joins = []
235: 
236:       joins = @joins_values.map {|j| j.respond_to?(:strip) ? j.strip : j}.uniq
237:

239:         association_joins << join if [Hash, Array, Symbol].include?(join.class) && !array_of_strings?(join)
240:       end
241: 
242:       stashed_association_joins = joins.grep(ActiveRecord::Associations::ClassMethods::JoinDependency::JoinAssociation)
243: 
244:       non_association_joins = (joins - association_joins - stashed_association_joins)
245:       custom_joins = custom_join_sql(*non_association_joins)
246: 
247:       join_dependency = ActiveRecord::Associations::ClassMethods::JoinDependency.new(@klass, association_joins, custom_joins)
248: 
249:       join_dependency.graft(*stashed_association_joins)
250: 
251:       @implicit_readonly = true unless association_joins.empty? && stashed_association_joins.empty?
252: 
253:       to_join = []
254: 
255:       join_dependency.join_associations.each do |association|
256:         if (association_relation = association.relation).is_a?(Array)
257:           to_join << [association_relation.first, association.join_type, association.association_join.first]
258:           to_join << [association_relation.last, association.join_type, association.association_join.last]

200:     def collapse_wheres(arel, wheres)
201:       equalities = wheres.grep(Arel::Nodes::Equality)
202: 
203:       groups = equalities.group_by do |equality|
204:         equality.left
205:       end
206: 
207:       groups.each do |_, eqls|
208:         test = eqls.inject(eqls.shift) do |memo, expr|
209:           memo.or(expr)
210:         end
211:         arel = arel.where(test)
212:       end
213: 
214:       (wheres - equalities).each do |where|
215:         where = Arel.sql(where) if String === where
216:         arel = arel.where(Arel::Nodes::Grouping.new(where))
217:       end

219:     end
