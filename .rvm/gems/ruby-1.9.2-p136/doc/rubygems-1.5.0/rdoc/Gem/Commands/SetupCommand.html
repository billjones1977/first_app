Gem::Commands::SetupCommand

Installs RubyGems itself. This command is ordinarily only available from a RubyGems checkout or tarball.

260:   def make_destination_dirs(install_destdir)
261:     lib_dir = nil
262:     bin_dir = nil
263: 
264:     prefix = options[:prefix]
265:     site_or_vendor = options[:site_or_vendor]
266: 
267:     if prefix.empty? then
268:       lib_dir = Gem::ConfigMap[site_or_vendor]
269:       bin_dir = Gem::ConfigMap[:bindir]
270:     else
271:       # Apple installed RubyGems into libdir, and RubyGems <= 1.1.0 gets
272:       # confused about installation location, so switch back to
273:       # sitelibdir/vendorlibdir.
274:       if defined?(APPLE_GEM_HOME) and
275:         # just in case Apple and RubyGems don't get this patched up proper.
276:         (prefix == Gem::ConfigMap[:libdir] or
277:          # this one is important
278:          prefix == File.join(Gem::ConfigMap[:libdir], 'ruby')) then
279:          lib_dir = Gem::ConfigMap[site_or_vendor]
280:          bin_dir = Gem::ConfigMap[:bindir]
281:       else
282:         lib_dir = File.join prefix, 'lib'
283:         bin_dir = File.join prefix, 'bin'
284:       end
285:     end
286: 
287:     unless install_destdir.empty? then
288:       lib_dir = File.join install_destdir, lib_dir.gsub(/^[a-zA-Z]:/, '')
289:       bin_dir = File.join install_destdir, bin_dir.gsub(/^[a-zA-Z]:/, '')
290:     end
291: 
292:     mkdir_p lib_dir
293:     mkdir_p bin_dir
294: 
295:     return lib_dir, bin_dir
296:   end
